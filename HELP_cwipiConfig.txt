// The following pieces of information need to be provided in the file cwipiConfig to make the test case
// work. Here we provide a help with the meaning of each of them:

cwipiStep         // Number of time steps between each DA phase (int)
cwipiMembers      // Number of members in the ensemble (int)
subdomains        // Number of subdomains in parallel runs. It needs to be the same number as the one
                  // specified in the file decomposeParDict (int)
cwipiObsU         // Number of observation probes for one component of the velocity field (int)
cwipiObsp         // Number of observation probes for pressure (int)
cwipiObsCf        // Number of observation probes for friction coefficient (default = 0)(int)
cwipiParams       // Number of model's parameters to optimize with the DA (int)
geom_tol          // Geometric tolerance of the cwipi coupling meshes (default = 1)(double)
cwipiOutputNb     // Number of txt files printing the complete system state beginning from the last time (int)
sigmaUserU        // sigma of the EnKF (pertubation of the observation and diagonal of R matrix for velocity)(double)
sigmaUserp        // sigma of the EnKF (pertubation of the observation and diagonal of R matrix for pressure)(double)
sigmaUserCf       // sigma of the EnKF (pertubation of the observation and diagonal of R matrix for friction coefficient)(double)
cwipiVerbose      // Print all the debugging messages or not: 1 = printed, 0 = nothing (bool)
cwipiTimedObs     // Switch to for the obs: 1 = obs depends on time, 0 = obs does not depend on time (bool)
obsTimeStep       // The time step of the observations if the case is unsteady (double)
cwipiParamsObs    // Definition of observation parameter : 0 = vel, 1 = pres, 2 = both, 3 = vel+cf (int)
stateInfl         // State inflation lambda - 1 (default = 0)(double)
paramsInfl        // Parameters inflation lambda - 1 (double)
typeInfl          // Definition of inflation (0 = stochastic, 1 = deterministic) (bool)
clippingSwitch    // Switch for the clipping. If true, a file called "clippingCells.txt" containing the cells IDs of the
                  // clipping must be included in the root directory (bool)
localSwitch       // Switch for the localisation (basic clipping or hyperlocalisation activated)(bool)
hyperlocSwitch    // Switch for hyperlocalization. If true, a file called "clippingCells.txt" containing the cells IDs of all
                  // the clippings must be included in the root directory (bool)
paramEstSwitch    // Switch for parameter estimation (default = 1) (bool)
stateEstSwitch    // Switch for state estimation (default = 1) (bool)
Ux                // Specification if Ux is read or not (cwipiParamsObs needs to be either 0, 2 or 3)(bool)
Uy                // Specification if Uy is read or not (cwipiParamsObs needs to be either 0, 2 or 3)(bool)
Uz                // Specification if Uz is read or not (cwipiParamsObs needs to be either 0, 2 or 3)(bool)
typeInputs        // Inputs for R are given in absolute values (0), percentage (1) or potential function (2) (default = 0)(int)
sigmaLocX         // eta of the EnKF (pertubation of the Kalman gain to take into consideration the localization in X direction)(double)
sigmaLocY         // eta of the EnKF (pertubation of the Kalman gain to take into consideration the localization in Y direction)(double)
sigmaLocZ         // eta of the EnKF (pertubation of the Kalman gain to take into consideration the localization in Z direction)(double)
epsilon           // Value used to calculate the NSRMD (double)
sigmaUserUa       // sigma of the EnKF given by a+by^c (only if typeInputs = 2) (double)
sigmaUserUb       // sigma of the EnKF given by a+by^c (only if typeInputs = 2) (double)
sigmaUserUc       // sigma of the EnKF given by a+by^c (only if typeInputs = 2) (double)
